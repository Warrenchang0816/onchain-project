-- ================================================
-- 餐廳即期食物販售平台 - 資料庫建置 SQL
-- PostgreSQL
-- ================================================

-- 1. 商品表
CREATE TABLE IF NOT EXISTS products (
    id                  BIGSERIAL PRIMARY KEY,
    product_id          VARCHAR(64)    NOT NULL UNIQUE,
    merchant_address    VARCHAR(42)    NOT NULL,
    customer_address    VARCHAR(42),
    name                VARCHAR(255)   NOT NULL,
    description         TEXT,
    category            VARCHAR(64),
    image_url           TEXT,
    original_price      NUMERIC(18,6)  NOT NULL,
    discounted_price    NUMERIC(18,6)  NOT NULL,
    discount_rate       INT            NOT NULL DEFAULT 0,
    quantity            INT            NOT NULL DEFAULT 0,
    available_qty       INT            NOT NULL DEFAULT 0,
    status              VARCHAR(20)    NOT NULL DEFAULT 'AVAILABLE',
    -- AVAILABLE | RESERVED | SOLD | EXPIRED
    expiry_time         TIMESTAMPTZ,
    sales_target        INT            NOT NULL DEFAULT 0,
    sold_quantity       INT            NOT NULL DEFAULT 0,
    chain_id            VARCHAR(20),
    vault_address       VARCHAR(42),
    contract_product_id VARCHAR(64),
    payment_status      VARCHAR(20)    NOT NULL DEFAULT 'PENDING',
    created_at          TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    listed_at           TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    sold_out_at         TIMESTAMPTZ
);

-- 2. 訂單表
CREATE TABLE IF NOT EXISTS orders (
    id                  BIGSERIAL PRIMARY KEY,
    order_id            VARCHAR(64)    NOT NULL UNIQUE,
    product_id          BIGINT         NOT NULL REFERENCES products(id),
    customer_address    VARCHAR(42)    NOT NULL,
    merchant_address    VARCHAR(42)    NOT NULL,
    quantity            INT            NOT NULL,
    unit_price          NUMERIC(18,6)  NOT NULL,
    total_price         NUMERIC(18,6)  NOT NULL,
    status              VARCHAR(20)    NOT NULL DEFAULT 'PLACED',
    -- PLACED | CONFIRMED | COMPLETED | CANCELLED
    payment_tx_hash     VARCHAR(66),
    created_at          TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    completed_at        TIMESTAMPTZ
);

-- 3. 區塊鏈操作紀錄表
CREATE TABLE IF NOT EXISTS task_blockchain_logs (
    id               BIGSERIAL PRIMARY KEY,
    task_id          VARCHAR(64)  NOT NULL,
    action           VARCHAR(64)  NOT NULL,
    tx_hash          VARCHAR(66),
    chain_id         VARCHAR(20),
    contract_address VARCHAR(42),
    status           VARCHAR(20),
    created_at       TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 4. 任務表（原有功能）
CREATE TABLE IF NOT EXISTS tasks (
    id                      BIGSERIAL PRIMARY KEY,
    task_id                 VARCHAR(64)    NOT NULL UNIQUE,
    wallet_address          VARCHAR(42)    NOT NULL,
    assignee_wallet_address VARCHAR(42),
    title                   VARCHAR(255)   NOT NULL,
    description             TEXT,
    status                  VARCHAR(20)    NOT NULL DEFAULT 'OPEN',
    priority                VARCHAR(20)    NOT NULL DEFAULT 'MEDIUM',
    reward_amount           NUMERIC(18,6),
    fee_bps                 INT            DEFAULT 0,
    chain_id                VARCHAR(20),
    vault_contract_address  VARCHAR(42),
    contract_task_id        VARCHAR(64),
    onchain_status          VARCHAR(20),
    fund_tx_hash            VARCHAR(66),
    approve_tx_hash         VARCHAR(66),
    claim_tx_hash           VARCHAR(66),
    cancel_tx_hash          VARCHAR(66),
    due_date                TIMESTAMPTZ,
    created_at              TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    updated_at              TIMESTAMPTZ    NOT NULL DEFAULT NOW()
);

-- 5. SIWE Nonce 表（錢包登入用）
CREATE TABLE IF NOT EXISTS auth_nonce (
    id             BIGSERIAL PRIMARY KEY,
    wallet_address VARCHAR(42)  NOT NULL,
    nonce          VARCHAR(128) NOT NULL,
    issued_at      TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    expired_at     TIMESTAMPTZ  NOT NULL,
    used           BOOLEAN      NOT NULL DEFAULT FALSE,
    created_at     TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at     TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- 6. 錢包 Session 表（登入狀態）
CREATE TABLE IF NOT EXISTS wallet_session (
    session_token  VARCHAR(128) PRIMARY KEY,
    wallet_address VARCHAR(42)  NOT NULL,
    chain_id       VARCHAR(20),
    expired_at     TIMESTAMPTZ  NOT NULL,
    revoked        BOOLEAN      NOT NULL DEFAULT FALSE,
    created_at     TIMESTAMPTZ  NOT NULL DEFAULT NOW(),
    updated_at     TIMESTAMPTZ  NOT NULL DEFAULT NOW()
);

-- ================================================
-- 索引
-- ================================================
CREATE INDEX IF NOT EXISTS idx_products_merchant    ON products(merchant_address);
CREATE INDEX IF NOT EXISTS idx_products_status      ON products(status);
CREATE INDEX IF NOT EXISTS idx_products_expiry      ON products(expiry_time);
CREATE INDEX IF NOT EXISTS idx_orders_customer      ON orders(customer_address);
CREATE INDEX IF NOT EXISTS idx_orders_merchant      ON orders(merchant_address);
CREATE INDEX IF NOT EXISTS idx_orders_product       ON orders(product_id);
CREATE INDEX IF NOT EXISTS idx_orders_status        ON orders(status);
CREATE INDEX IF NOT EXISTS idx_auth_nonce_wallet    ON auth_nonce(wallet_address);
CREATE INDEX IF NOT EXISTS idx_wallet_session_wallet ON wallet_session(wallet_address);
